29def prepare_sequences(notes, pitchnames, n_vocab): 

30 """ Prepare the sequences used by the Neural Network """ 

31 # map between notes and integers and back 

32 note_to_int = dict((note, number) for number, note in enumerate(pitchnames)) 

33 

34 sequence_length = 100 

35 network_input = [] 

36 output = [] 

37 for i in range(0, len(notes) - sequence_length, 1): 

38 sequence_in = notes[i:i + sequence_length] 

39 sequence_out = notes[i + sequence_length] 

40 network_input.append([note_to_int[char] for char in sequence_in]) 

41 output.append(note_to_int[sequence_out]) 

42 

43 n_patterns = len(network_input) 

44 

45 # reshape the input into a format compatible with LSTM layers 

46 normalized_input = numpy.reshape(network_input, (n_patterns, sequence_length, 1)) 

47 # normalize input 

48 normalized_input = normalized_input / float(n_vocab) 

49 

50 return (network_input, normalized_input)

52def create_network(network_input, n_vocab,weights): 

53 """ create the structure of the neural network """ 

54 model = Sequential() 

55 model.add(LSTM( 

56 512, 

57 input_shape=(network_input.shape[1], network_input.shape[2]), 

58 recurrent_dropout=0.3, 

59 return_sequences=True 

60 )) 

61 model.add(LSTM(512, return_sequences=True, recurrent_dropout=0.3,)) 

62 model.add(LSTM(512)) 

63 model.add(BatchNorm()) 

64 model.add(Dropout(0.3)) 

65 model.add(Dense(256)) 

66 model.add(Activation('relu')) 

67 model.add(BatchNorm()) 

68 model.add(Dropout(0.3)) 

69 model.add(Dense(n_vocab)) 

70 model.add(Activation('softmax')) 

71 model.compile(loss='categorical_crossentropy', optimizer='rmsprop') 

72 

73 # Load the weights to each node

104def create_midi(prediction_output): 

105 """ convert the output from the prediction to notes and create a midi file 

106 from the notes """ 

107 offset = 0 

108 output_notes = [] 

109 

110 # create note and chord objects based on the values generated by the model 

111 for pattern in prediction_output: 

112 # pattern is a chord 

113 if ('.' in pattern) or pattern.isdigit(): 

114 notes_in_chord = pattern.split('.') 

115 notes = [] 

116 for current_note in notes_in_chord: 

117 new_note = note.Note(int(current_note)) 

118 new_note.storedInstrument = instrument.Piano() 

119 notes.append(new_note) 

120 new_chord = chord.Chord(notes) 

121 new_chord.offset = offset 

122 output_notes.append(new_chord) 

123 # pattern is a note 

124 else: 

125 new_note = note.Note(pattern) 

126 new_note.offset = offset 

127 new_note.storedInstrument = instrument.Piano() 

128 output_notes.append(new_note) 

129 

130 # increase offset each iteration so that notes do not stack 

131 offset += 0.5 

132 

133 midi_stream = stream.Stream(output_notes) 

134 

135 midi_stream.write('midi', fp='test_output2.mid')
